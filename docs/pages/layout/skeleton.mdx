# Skeleton

> A utility to create placeholders via [react content loader](https://skeletonreact.com/)

<Theme>
  {theme => (
    <Skeleton
      speed={2}
      width={340}
      height={84}
      viewBox="0 0 340 84"
      backgroundColor={theme === 'dark' ? '#333' : '#999'}
      foregroundColor={theme === 'dark' ? '#666' : '#CCC'}
    />
  )}
</Theme>
